2.1 info Lambda иЎЁиҫҫејҸе®ҡд№ү

Lambda иЎЁиҫҫејҸе®ҡд№ү

  • Lambda иЎЁиҫҫејҸ пјҡиҝҷжҳҜ Java8 дёӯжңҖе…·д»ЈиЎЁжҖ§зҡ„иҜӯжі•зі–пјҢе®ғе…Ғи®ёдҪ е°ҶеҮҪж•°дҪңдёәж–№жі•еҸӮж•°дј йҖ’пјҢжҲ–иҖ…е°Ҷд»Јз ҒеғҸж•°жҚ®дёҖж ·еӨ„зҗҶгҖӮLambda иЎЁиҫҫејҸеҸҜд»ҘжһҒеӨ§ең°з®ҖеҢ–еҢҝеҗҚеҶ…йғЁзұ»зҡ„еҶҷжі•гҖӮ

  • иҜӯжі•зі– пјҡ иҜӯжі•зі–пјҲSyntactic SugarпјүжҳҜжҢҮеңЁи®Ўз®—жңәиҜӯиЁҖдёӯж·»еҠ зҡ„жҹҗз§ҚиҜӯжі•пјҢиҝҷз§ҚиҜӯжі•еҜ№иҜӯиЁҖзҡ„еҠҹиғҪе№¶жІЎжңүеҪұе“ҚпјҢдҪҶжҳҜжӣҙж–№дҫҝзЁӢеәҸе‘ҳдҪҝз”ЁгҖӮе®ғеҸҜд»Ҙи®©д»Јз ҒжӣҙеҠ з®ҖжҙҒгҖҒжҳ“иҜ»пјҢжҸҗеҚҮејҖеҸ‘ж•ҲзҺҮ

LambdaиЎЁиҫҫејҸзҡ„жң¬иҙЁ е’Ңи®ҫи®Ўзӣ®ж Ү

  • жң¬иҙЁ пјҡеҢҝеҗҚеҮҪж•° зҡ„з®Җд»ӢиЎЁзӨәжі•пјҢз”ЁдәҺе®һзҺ°еҮҪж•°ејҸжҺҘеҸЈ (д»…еҗ«1дёӘжҠҪиұЎж–№жі•зҡ„жҺҘеҸЈ)
  • и®ҫи®Ўзӣ®ж Ү : з®ҖеҢ–еҢҝеҗҚеҶ…йғЁзұ»еҶҷжі•пјҢж”ҜжҢҒеҮҪж•°ејҸзј–зЁӢиҢғејҸ

2.2 Lambda иЎЁиҫҫејҸзҡ„иҜӯжі•

2.2.1 иҜӯжі•

(parameters) -> {expression_or_statements}

2.2.2 иҜӯжі•жһ„жҲҗи§Јжһҗ

з»„жҲҗйғЁеҲҶиҜҙжҳҺ
parametersеҪўеҸӮеҲ—иЎЁпјҲеҸҜзңҒз•Ҙзұ»еһӢеЈ°жҳҺпјҢеҚ•еҸӮж•°ж—¶еҸҜзңҒз•ҘжӢ¬еҸ·пјү
->Lambdaж“ҚдҪңз¬ҰпјҲз®ӯеӨҙз¬ҰеҸ·пјү
expressionеҚ•жқЎиЎЁиҫҫејҸпјҲеҸҜзңҒз•Ҙ{}е’Ңreturnпјү
statementsд»Јз Ғеқ—пјҲеӨҡжқЎиҜӯеҸҘйңҖ{}еҢ…иЈ№пјҢжҳҫејҸreturnиҝ”еӣһеҖјпјү

2.3 Lambda иЎЁиҫҫејҸзҡ„жј”иҝӣиҝҮзЁӢ

дёӢйқўжҲ‘д»¬д»Һд»Јз Ғи§’еәҰпјҢ1дёӘзұ» д»ҺдёҠеҲ°дёӢзҡ„жү§иЎҢйЎәеәҸпјҢд»ҘеҸҠжіЁйҮҠжқҘж…ўж…ўжј”иҝӣLambda иЎЁиҫҫејҸ

package com.learn.stream.api.lambda;

// д»ҘеүҚеҶҷдёҖдёӘжҺҘеҸЈ
interface MyInterface {
    int sum(int a, int b);
}

interface MyNoParamInterface {
    Integer no();
}

// з„¶еҗҺеҶҷдёҖдёӘе®ғзҡ„е®һзҺ°зұ»
class MyImpl implements MyInterface {
    @Override
    public int sum(int a, int b) {
        return a + b;
    }
}

/**
 * еӯҰд№ LambdaиЎЁиҫҫејҸ
 *
 * @author qianpengzhan
 * @since 2025/6/27 17:02
 */
public class LambdaLearn {

    public static void main(String[] args) {
        //1. иҮӘе·ұеҲӣе»әе®һзҺ°зұ»еҺ»е®һзҺ°
        MyInterface myInterface = new MyImpl();
        System.out.println(myInterface.sum(10, 20));

        //2. еҲӣе»әеҢҝеҗҚе®һзҺ°зұ»
        MyInterface myInterface1 = new MyInterface() {
            @Override
            public int sum(int a, int b) {
                return a * b;
            }
        };
        System.out.println(myInterface1.sum(10, 20));
        // еҸҜд»ҘеҲ°дёҠиҝ°еҢҝеҗҚе®һзҺ°зұ»зҡ„еҶҷжі• йқһеёёзҡ„еҶ—дҪҷ ---> е°ұиҝҷж · еј•еҮәдәҶ lambdaиЎЁиҫҫејҸ

        //3. дҪҝз”ЁlambdaиЎЁиҫҫејҸ йӮЈд№ҲжҖҺд№ҲеҶҷе‘ўпјҹ е®һйҷ…дёҠе°ұи§үеҫ—еҶ—дҪҷзҡ„еҺ»жҺү жҜ”еҰӮ new MyInterface { е’Ң }
        // д»ҘеҸҠ @Overide public int sumиҝҷдәӣзұ»еһӢгҖҒдҝ®йҘ°з¬ҰгҖҒж–№жі•еҗҚз§°йғҪжҳҜдёҚеҸҜеҸҳзҡ„дёңиҘҝ еҺ»жҺү  з„¶еҗҺ дҪҝз”ЁеҠЁжҖҒзҡ„еҶҷжі• еҠ дёҠж“ҚдҪңз¬Ұ ->
        MyInterface myInterface2 = (int a, int b) -> {
            return a * b;
        };
        System.out.println(myInterface2.sum(10, 20));
        // д»ҺдёҠиҝ°д»Јз ҒеҸҜд»ҘзңӢеҲ° е…¶е®һ lambdaиЎЁиҫҫејҸе°ұжҳҜ  еҸӮж•°еҲ—иЎЁ + ж“ҚдҪңз¬ҰеҸ·(->) + ж–№жі•дҪ“
        // еҗҢж ·жҲ‘д»¬д№ҹзңӢеҮәжқҘ жҲ‘д»¬еҸӘиҰҒдјҡеҶҷеҢҝеҗҚе®һзҺ°зұ» иҮӘз„¶е°ұдјҡеҶҷ lambda иЎЁиҫҫејҸдәҶ
        // дёҠиҝ°д»Јз Ғ е…¶е®һиҝҳеҸҜд»ҘеҶҚж¬Ўз®ҖеҢ– дёҠиҝ°дёә1дёӘе®Ңж•ҙзҡ„еҶҷжі•

        // 3.1 з®ҖеҢ–еҶҷжі•дёҖ еҸӮж•°зұ»еһӢеҸҜд»ҘдёҚеҶҷ  жҺҘеҸЈе®ҡд№ү еҸӮж•°зұ»еһӢжҳҜеӣәе®ҡзҡ„. иҖҢдё”еҸӮж•°еҸҳйҮҸеҗҚд№ҹжҳҜеҸҜд»Ҙж”№зҡ„пјҢйҡҸж„Ҹе®ҡд№ү
        MyInterface myInterface3 = (a, b) -> {
            return a * a + b * b;
        };
        System.out.println(myInterface3.sum(10, 20));

        // 3.2 з®ҖеҢ–еҶҷжі•дәҢ жңҖе°‘еҸӮж•°еҪўејҸ  еӣ дёәдёҠиҝ°жҳҜеӣәе®ҡдәҶ2дёӘеҸӮж•°пјҢжңүдәӣж— еҸӮж–№жі•е°ұеҸҜд»ҘеҰӮдёӢеҶҷжі•
        // еҺ»дёҠйқўе®ҡд№ү1дёӘж— еҸӮж–№жі•зҡ„жҺҘеҸЈ
        MyNoParamInterface myNoParamInterface = () -> {
            return 10;
        };
        System.out.println(myNoParamInterface.no());

        // 3.3 з®ҖеҢ–еҶҷжі•дёү  ж–№жі•дҪ“иӢҘжҳҜеҸӘжңү1иЎҢд»Јз Ғ {}йғҪеҸҜд»ҘзңҒз•Ҙ
        MyNoParamInterface myNoParamInterface1 = () -> 20;
        System.out.println(myNoParamInterface1.no());

        // 4.з»јдёҠжүҖиҝ°, lambdaиЎЁиҫҫејҸз®ҖеҢ–дәҶеҫҲеӨҡд»Јз ҒгҖӮ
        // еңЁjavaдёӯ,еҸӘиҰҒжҳҜеҮҪж•°ејҸжҺҘеҸЈ,е°ұеҸҜд»ҘдҪҝз”ЁlambdaиЎЁиҫҫејҸгҖӮ
    }
}

3.2.1 еҚ•жҠҪиұЎж–№жі•еҺҹеҲҷ

package java.util.function;
import java.util.Objects;
@FunctionalInterface
public interface Function<T, R> {
    /**
     * иҫ“е…Ҙ1дёӘеҸӮж•°, иҝ”еӣһ1дёӘз»“жһңеҸӮж•°
     */
    R apply(T t);
    default <V> Function<V, R> compose(Function<? super V, ? extends T> before) {
        Objects.requireNonNull(before);
        return (V v) -> apply(before.apply(v));
    }
    default <V> Function<T, V> andThen(Function<? super R, ? extends V> after) {
        Objects.requireNonNull(after);
        return (T t) -> after.apply(apply(t));
    }
    static <T> Function<T, T> identity() {
        return t -> t;
    }
}

еҸҜд»ҘеҸ‘зҺ°пјҢе®ғжҳҜдёҖдёӘеҮҪж•°ејҸжҺҘеҸЈпјҢеҢ…еҗ«дёҖдёӘжҠҪиұЎж–№жі• apply(T t)пјҢжҺҘеҸ—зұ»еһӢдёә T зҡ„еҸӮж•°пјҢиҝ”еӣһзұ»еһӢдёә R зҡ„з»“жһңвҖҢпјҢз„¶еҗҺеҢ…еҗ«composeгҖҒandThenгҖҒidentity 3дёӘй»ҳи®Өж–№жі•гҖӮ

        ToDoubleBiFunction<Double, Double> toDoubleBiFunction = Double::sum;

        Predicate<Integer> predicate = x -> x > 3;
